$ pip install mcp-context-sdk 
from mcp_context_sdk import ContextBuilder

# Create a context
context = ContextBuilder()\
    .with_schema("coding")\
    .with_version("v1")\
    .build()

# Use with your AI system
response = ai_model.generate(
    context=context
)
